cmake_minimum_required(VERSION 3.0)
project(kissfft)
set(SRCS "kiss_fft.c" "tools/kiss_fftr.c")
include_directories(".")
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fPIC")
set(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS} -O3 -DNDEBUG")
set(CMAKE_C_FLAGS_DEBUG "${CMAKE_C_FLAGS} -Og -g3 -DDEBUG")
add_library(kissfft STATIC ${SRCS})
install(TARGETS kissfft DESTINATION "lib")
install(FILES "kiss_fft.h" "tools/kiss_fftr.h" DESTINATION "include")
